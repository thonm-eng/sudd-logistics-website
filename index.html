-- Shipments table (for admin CRUD + public tracking) create table if not exists shipments ( id bigint generated always as identity primary key, created_at timestamptz default now(), tracking_number text not null unique, bl_number text, consignee text, vessel text, shipping_line text, container_size text default '40ft', origin text, destination text, current_stage int default 0, eta_note text, stage_log text ); -- Profiles table (populated when clients sign up) create table if not exists profiles ( id uuid primary key, created_at timestamptz default now(), full_name text, email text, company_name text, phone text ); -- RLS policies alter table shipments enable row level security; alter table profiles enable row level security; -- Public can read shipments (for tracking page) create policy "Public can read shipments" on shipments for select using (true); -- Only authenticated admin can insert/update/delete shipments create policy "Admin manages shipments" on shipments for all using (auth.role() = 'authenticated'); -- Profiles readable by owner and admin create policy "Owner reads own profile" on profiles for select using (auth.uid() = id); create policy "Admin reads all profiles" on profiles for select using (auth.role() = 'authenticated'); create policy "Owner inserts own profile" on profiles for insert with check (auth.uid() = id); create policy "Owner updates own profile" on profiles for update using (auth.uid() = id); create policy "Admin deletes profiles" on profiles for delete using (auth.role() = 'authenticated');
